procedure BinarySearch(A, target)
    left ← 0
    right ← length(A) − 1

    while left ≤ right do
        mid ← floor((left + right) / 2)

        if A[mid] = target then
            return mid                 // Found at index mid
        else if A[mid] < target then
            left ← mid + 1             // Search right half
        else
            right ← mid − 1            // Search left half
        end if
    end while

    return -1                          // Not found
end procedure
